227 if (strstr(service_type_string, "display") != NULL)

232 return "Communication";
233 if (strstr(service_type_string, "wifi") != NULL ||
234 strstr(service_type_string, "mqtt") != NULL ||
235 strstr(service_type_string, "http") != NULL ||
236 strstr(service_type_string, "ble") != NULL)
237 return "Network";
